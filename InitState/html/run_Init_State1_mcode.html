1function [msg, likelihoods, templates, vanishingPt] = run_Init_State(RES_VH, NBUFFER, laneFilter, vpFilter)
  2    
  3  coder.cstructname(laneFilter, 'MatlabStruct_laneFilter');
  4  coder.cstructname(vpFilter, 'MatlabStruct_vpFilter');
  5        
  6    %%%%%%%%%%%%%%%%%%%%%%%%%%%
  7    %% (re)start the filters %%
  8    %%%%%%%%%%%%%%%%%%%%%%%%%%%
  9   laneFilter.mFilter = laneFilter.mPrior;
 10   vpFilter.mFilter    = vpFilter.mPrior;
 11   
 12   
 13    
 14   
 15    
 16    
 17    TOT_P_ALL      = zeros( RES_VH(1), RES_VH(2), NBUFFER);
 18    DIR_ALL        = zeros( RES_VH(1), RES_VH(2), NBUFFER);
 19    TOT_P_ALL_BACK_UP = zeros( RES_VH(1), RES_VH(2), NBUFFER);
 20    DIR_ALL_BACK_UP   = zeros( RES_VH(1), RES_VH(2), NBUFFER);
 21    MASK_FOC_TOT_P = ones(  RES_VH(1), RES_VH(2));
 22    FOC_TOT_P      = zeros( RES_VH(1), RES_VH(2));
 23    TOT_P          = zeros( RES_VH(1), RES_VH(2));
 24    AVG_DIR_TOT_P  = zeros( RES_VH(1), RES_VH(2));
 25    IDX_LANE_PIX   = [];
 26    
 27    %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
 28    %% create the gradient template %%
 29    %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
 30    ROOT_DIR_TEMPLATE = createTemplate(RES_VH(1),RES_VH(2));
 31    SEGMENT           = zeros(RES_VH(1),RES_VH(2));
 32
 33    %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
 34    %% create the prob template               %%
 35    %% masks out parts above horizon + margin %%
 36    %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
 37    Margin          = 100; %% Note: Should be fitted using sweeps
 38    ROOT_PROB_TEMPLATE = zeros(2*RES_VH(1)+1,RES_VH(2));
 39    ROOT_PROB_TEMPLATE(RES_VH(1)+Margin:end,:) = 1;
 40    for n = 481:RES_VH(1)+Margin
 41      ROOT_PROB_TEMPLATE(n,:) = ((n-RES_VH(1))/Margin)^3;
 42    end
 43    
 44    %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
 45    %% create the depth template              %%
 46    %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
 47    ROOT_IDEPTH_TEMPLATE = zeros(2*RES_VH(1)+1,RES_VH(2));
 48    step  = 45/RES_VH(1);
 49    angle = 90-step;
 50    for n = RES_VH(1)+1:2*RES_VH(1)+1
 51        ROOT_IDEPTH_TEMPLATE(n,:) = 1.75*tand(angle);
 52        angle                     = angle-step;
 53    end
 54    ROOT_IDEPTH_TEMPLATE( 100 < ROOT_IDEPTH_TEMPLATE ) = 100;
 55    
 56%     figure(234)
 57%     imshow(ROOT_IDEPTH_TEMPLATE,[0 100])
 58%     colormap jet
 59%     drawnow
 60%     pause
 61    
 62    
 63
 64    %%%%%%%%%%%%%%%%%%%%%%%%
 65    %% this state is done %%
 66    %%%%%%%%%%%%%%%%%%%%%%%%
 67    
 68                      
 69                     templates = struct;
 70                          templates.ROOT_DIR_TEMPLATE = ROOT_DIR_TEMPLATE;
 71                          templates.ROOT_PROB_TEMPLATE = ROOT_PROB_TEMPLATE;
 72                          templates.ROOT_PROB_TEMPLATE = ROOT_PROB_TEMPLATE;
 73                          templates.ROOT_DEPTH_TEMPLATE = ROOT_IDEPTH_TEMPLATE;
 74                          templates.SEGMENT = SEGMENT;
 75                      
 76                     
 77                     vanishingPt = struct;
 78                         vanishingPt.V = 0;
 79                         vanishingPt.H = 0;
 80                         vanishingPt.V_prev = 0;
 81                         vanishingPt.H_prev = 0;
 82
 83
 84                    likelihoods =struct;
 85                          likelihoods.TOT_P_ALL         = TOT_P_ALL;
 86                          likelihoods.DIR_ALL           = DIR_ALL;
 87                          likelihoods.MASK_FOC_TOT_P    = MASK_FOC_TOT_P;
 88                          likelihoods.TOT_P             = TOT_P;
 89                          likelihoods.FOC_TOT_P         = FOC_TOT_P;
 90                          likelihoods.AVG_DIR_TOT_P     = AVG_DIR_TOT_P;
 91                          likelihoods.TOT_P_ALL_BACK_UP = TOT_P_ALL_BACK_UP;
 92                          likelihoods.DIR_ALL_BACK_UP   = DIR_ALL_BACK_UP;
 93                          likelihoods.IDX_LANE_PIX      = IDX_LANE_PIX;
 94                          
 95                          
 96    coder.cstructname(templates, 'MatlabStruct_templates');
 97    coder.cstructname(vanishingPt, 'MatlabStruct_vanishingPt');
 98    coder.cstructname(likelihoods, 'MatlabStruct_likelihoods');
 99               
100    
101

103    
104end
